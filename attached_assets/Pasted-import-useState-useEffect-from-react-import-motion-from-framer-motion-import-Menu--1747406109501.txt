function Navbar() {
  const [open, setOpen] = useState(false);
  const [scrolled, setScrolled] = useState(false);
  
  useEffect(() => {
    const handleScroll = () => {
      if (window.scrollY > 50) {
        setScrolled(true);
      } else {
        setScrolled(false);
      }
    };
    
    window.addEventListener('scroll', handleScroll);
    return () => {
      window.removeEventListener('scroll', handleScroll);
    };
  }, []);

  return (
    <motion.nav 
      className={`fixed z-50 top-0 left-0 w-full ${scrolled ? 'bg-white/95 shadow-lg backdrop-blur-sm' : 'bg-transparent'} px-4 py-3 flex justify-between items-center transition-all duration-300`}
      initial={{ y: -100 }}
      animate={{ y: 0 }}
      transition={{ duration: 0.5 }}
    >
      <span className="font-bold text-2xl bg-gradient-to-r from-blue-600 to-purple-600 bg-clip-text text-transparent">
        Faberlic ДЭНАС
      </span>
      <div className="hidden lg:flex gap-6">
        {navLinks.map((link) => (
          <a
            href={link.href}
            key={link.label}
            className="font-medium text-gray-700 hover:text-blue-600 transition-colors"
          >
            {link.label}
          </a>
        ))}
        <a
          href="https://t.me/official_faberlic"
          className="bg-blue-500 text-white rounded-full px-4 py-2 flex items-center gap-1 hover:bg-blue-600 transition-colors"
          target="_blank"
          rel="noopener noreferrer"
        >
          <MessageSquare size={16} /> Telegram
        </a>
        <a
          href="https://faberlic.com/ru/ru"
          className="bg-indigo-500 text-white rounded-full px-4 py-2 hover:bg-indigo-600 transition-colors flex items-center gap-1"
          target="_blank"
          rel="noopener noreferrer"
        >
          <ShoppingCart size={16} /> Купить
        </a>
      </div>
      <button className="lg:hidden z-50" onClick={() => setOpen(!open)}>
        {open ? <X size={24} /> : <Menu size={24} />}
      </button>
      {open && (
        <motion.div
          initial={{ opacity: 0, x: "100%" }}
          animate={{ opacity: 1, x: 0 }}
          exit={{ opacity: 0, x: "100%" }}
          transition={{ duration: 0.3 }}
          className="absolute top-0 right-0 h-screen w-4/5 bg-white shadow-2xl p-8 flex flex-col gap-6 pt-20"
        >
          {navLinks.map((link) => (
            <a
              href={link.href}
              key={link.label}
              className="text-gray-800 text-lg font-medium hover:text-blue-600 transition-colors"
              onClick={() => setOpen(false)}
            >
              {link.label}
            </a>
          ))}
          <div className="mt-auto space-y-4">
            <a
              href="https://t.me/official_faberlic"
              className="bg-blue-500 text-white rounded-full px-4 py-3 flex items-center justify-center gap-2 hover:bg-blue-600 transition-colors"
              target="_blank"
              rel="noopener noreferrer"

              <MessageSquare size={18} /> Telegram
            </a>
            <a
              href="https://faberlic.com/ru/ru"
              className="bg-indigo-500 text-white rounded-full px-4 py-3 flex items-center justify-center gap-2 hover:bg-indigo-600 transition-colors"

function Catalog() {
  const [category, setCategory] = useState("all");
  const [searchTerm, setSearchTerm] = useState("");
  
  const categories = ["all", ...new Set(products.map(product => product.category))];
  
  const filteredProducts = products.filter(product => {
    const matchesCategory = category === "all" || product.category === category;
    const matchesSearch = product.title.toLowerCase().includes(searchTerm.toLowerCase()) || 
                        product.description.toLowerCase().includes(searchTerm.toLowerCase());
    return matchesCategory && matchesSearch;
  });
  
  return (
    <section id="catalog" className="py-20 bg-gray-50 px-4">
      <div className="max-w-6xl mx-auto">
        <motion.div 
          className="mb-12 text-center"
          initial={{ opacity: 0, y: 20 }}
          whileInView={{ opacity: 1, y: 0 }}
          viewport={{ once: true }}
          transition={{ duration: 0.6 }}
        >
          <h2 className="text-3xl lg:text-4xl font-bold bg-gradient-to-r from-blue-700 to-purple-700 bg-clip-text text-transparent mb-4">
            Каталог продукции
          </h2>
          <p className="text-lg text-gray-600 max-w-3xl mx-auto">
            Широкий ассортимент современных аппаратов ДЭНАС для вашего здоровья
          </p>
        </motion.div>
        
        <div className="mb-8 flex flex-col md:flex-row justify-between items-center gap-4">
          <div className="flex flex-wrap gap-2">
            {categories.map((cat) => (
              <button
                key={cat}
                onClick={() => setCategory(cat)}
                className={`px-4 py-2 rounded-full text-sm md:text-base transition-colors ${
                  category === cat
                    ? "bg-blue-600 text-white"
                    : "bg-white text-gray-700 hover:bg-blue-100"
                }`}
              >
                {cat === "all" ? "Все категории" : cat}
              </button>
            ))}
          </div>
          
          <div className="relative w-full md:w-64">
            <input
              type="text"
              placeholder="Поиск..."
              value={searchTerm}
              onChange={(e) => setSearchTerm(e.target.value)}
              className="w-full px-4 py-2 pl-10 rounded-full border border-gray-300 focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent"
            />
            <Search className="absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400" size={18} />

        </div>
        
        <div className="grid sm:grid-cols-2 lg:grid-cols-3 gap-8">

              <div className="p-6">
                <div className="text-xs font-medium text-blue-600 mb-2">{product.category}</div>
                <h3 className="text-xl font-bold text-gray-900 mb-2">{product.title}</h3>
